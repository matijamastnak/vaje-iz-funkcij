\chapter{Uvod}
\label{cha:uvod}

Tu bo en lep uvod.

\begin{zgled}
  Reši enačbo $x^2 + 4 x + 1 = 0$.
\end{zgled}

\begin{resitev}
  Najprej dopolnimo do kvadrata,
  %
  \begin{equation*}
    (x + 2)^2 - 3 = 0,
  \end{equation*}
  %
  in razberemo rešitvi:
  %
  \begin{equation*}
    x_{1,2} = -2 + \pm \sqrt{3}.
  \end{equation*}
\end{resitev}
